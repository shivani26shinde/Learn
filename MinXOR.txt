public class Solution {
    public int findMinXor(ArrayList<Integer> A) {
        Collections.sort(A);
        if(A.size() == 1){
            return A.get(0);
        }
        int min = Integer.MAX_VALUE;
        for(int i=0;i<A.size()-1;i++){
            min = Math.min(min, A.get(i)^A.get(i+1));
        }
        return min;
    }
}
